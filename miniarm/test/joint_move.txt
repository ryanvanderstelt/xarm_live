#include <Arduino.h>
#include <ESP32Servo.h>
#include "miniarm.h"


// Servo pins for each joint
const int SERVO_PINS[] = {
  21, // Joint 0: Base
  19, // Joint 1: Shoulder
  18, // Joint 2: Elbow
  27, // Joint 3: Wrist
  26  // Joint 4: Claw
};

// Servo objects
Servo motors[5];
const char* joint_names[] = {"Base", "Shoulder", "Elbow", "Wrist", "Claw"};

void setup() {
  Serial.begin(115200);
  delay(1000);
  RobotInitialize();  

}

void loop() {

  for (int angle = 0; angle <= 180; angle++){
    motors[]->write(angle);
    delay(15);
  }  
}
